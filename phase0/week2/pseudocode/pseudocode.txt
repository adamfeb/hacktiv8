// 1. Newton Second Law
//


// Algorithm

1. Buat sebuah variable dengan nama "mass" dan berikan nilai sebesar 600
2. Buat sebuah variable dengan nama "acceleration" dan berikan nilai sebesar 2
3. Buat sebuah variable dengan nama "force" tanpa nilai
4. Hitung "mass" dikali "acceleration"
5. Berikan nilai pada "force" sesuai dengan hasil perhitungan
6. Tampilkan nilai "force"


// PseudoCode

STORE "mass" with 600
STORE "acceleration" with 2
STORE "force" without any value

CALCULATE "mass" times "acceleration"
SET "force" value with calculation result

DISPLAY "force"





//
// 2. Tahun Kabisat
//


// Algorithm

1. Dapatkan nilai dan berikan nilai tersebut ke variable "tahun"
2. Jika sisa hasil bagi dari "tahun" dibagi 4 adalah 0 dan sisa hasil bagi dari "tahun" dibagi 100 adalah 1,
   atau jika sisa hasil bagi dari "tahun" dibagi 4 adalah 0 dan sisa hasil bagi dari "tahun" dibagi 100 adalah 0 dan sisa hasil bagi dari "tahun" dibagi 400 adalah 0 lalu
3. Tampilkan "Ini adalah Tahun Kabisat"
4. Selain itu tampilkan "Ini bukan Tahun Kabisat"


// PseudoCode

READ "year"

IF ("year" mod 4 is 0 AND "year" mod 100 is 1) OR ("year" mod 4 is 0 AND "year" mod 100 is 0 AND "year" mod 400 is 0) THEN
  WRITE "This is a Leap Year"
ELSE 
  WRITE "This is not a Leap Year"
ENDIF





//
// 3. Laundry Day
//


// Algorithm

1. Buat sebuah variable dengan nama "jumlahPakaian" dan berikan nilai sebesar 20
2. Atur nilai awal "perhitungan" sebesar 0
3. Selama "perhitungan" kurang dari "jumlahPakaian" lakukan
     Tambahkan 1 ke "perhitungan"
     Catat nilai "perhitungan"
     Lakukan "Masukan Pakaian Ke Mesin Cuci"
4. Setelah itu "Nyalakan Mesin Cuci"


// PseudoCode

STORE "clothes" with 20
SET initial "count" to 0

WHILE "count" < "clothes"
  ADD 1 to "count"
  WRITE "count"
  DO "Put Clothes in Washing Machine" 
ENDWHILE

DO "Turn On Washing Machine"





//
// 4. Periksa Kuku
//


// Algorithm

1. Buat sebuah variable dengan nama "murid" dan berikan nilai sebesar 40
2. Atur nilai awal "perhitungan" sebesar 0
3. Selama "perhitungan" kurang dari "jumlahMurid" lakukan
     Tambahkan 1 ke "perhitungan" 
     Catat nilai "perhitungan" 
     Lakukan "Pemeriksaan Kuku"
       Jika Kuku Murid Panjang, lalu Guru menghukum siswa tersebut
       Selain Itu, Guru memberikan pujian


// PseudoCode

STORE "students" with 40
SET initial "count" to 0

WHILE count < students
  ADD 1 to count
  WRITE count
  DO "Check the Student's Nails" 
    IF "Nails is Long" THEN
      DO "Give a Punishment to the Student"
    ELSE
      DO "Give a Compliment to the Student"
    ENDIF
ENDWHILE
